\section{Sec. 1 title}
\frame{
\frametitle{S. Equation}

\begin{itemize}
\item When $m=1$, the solution of NLS can be obtained through the
following minimization
\begin{align}\label{eq1.2}
\inf_{\tiny{\begin{array}{c}\phi\geq 0\\\phi\in H^{1}({\mathbb
R}^n)\end{array}}}\frac{\int_{{\mathbb R}^n}|\nabla
\phi|^2+\lambda\int_{{\mathbb R}^n}\phi^2}{\left(\int_{{\mathbb
R}^n}\phi^4\right)^{1/2}}.
\end{align}
\item An equivalent formulation is the following minimization:
\begin{subequations}\label{eq1.3}
\begin{align}\label{eq1.3a}
\inf_{\tiny{\phi\in \mathcal{N}_1}}E(\phi)
\end{align}
where \scriptsize{
\begin{align}\label{eq1.3b}
\mathcal{N}_1=\left\{\phi\in H^{1}({\mathbb R}^n)|\ \phi\geq
0,~\phi\equiv\!\!\!\!\!/ ~~0, \int_{{\mathbb R}^n}|\nabla
\phi|^2+\lambda\int_{{\mathbb R}^n}\phi^2= \mu\int_{{\mathbb
R}^n}\phi^4\right\}
\end{align}}
\begin{align}\label{eq1.3c}
E(\phi)=\frac{1}{2} \int_{{\mathbb R}^n}|\nabla
\phi|^2+\frac{\lambda}{2}\int_{{\mathbb R}^n}\phi^2-
\frac{\mu}{4}\int_{{\mathbb R}^n}\phi^4.
\end{align}
\end{subequations}
\item If $\phi$ satisfies \eqref{eq1.3} then $\phi$ is called a
ground state solution.
\end{itemize}
} % ---
